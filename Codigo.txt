// Seleccionar LegalSign
        const radioLegalSign = await esperarElemento('input[ng-reflect-value="P"]');
        radioLegalSign.click();
        radioLegalSign.dispatchEvent(new Event('change', { bubbles: true }));
        await pausa(300);
        
        // Inyectar RUT
        const inputRut = await esperarElemento('input.st-form-control__input');
        inputRut.focus(); 
        inputRut.value = rutLimpio;
        inputRut.dispatchEvent(new Event('input', { bubbles: true }));
        inputRut.dispatchEvent(new Event('change', { bubbles: true }));
        inputRut.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true, key: 'Enter' }));
        inputRut.blur();  
        inputRut.dispatchEvent(new Event('focusout', { bubbles: true })); // Extra para Angular
        
        console.log("⏳ Buscando el botón VISIBLE de Continuar...");
        await pausa(1000); 
        
        // --- BÚSQUEDA INTELIGENTE DE ELEMENTOS VISIBLES ---
        const todosLosBotones = document.querySelectorAll('button');
        let btnContinuar = null;
        for (let btn of todosLosBotones) {
            const textoBoton = btn.textContent.trim().toUpperCase();
            // Buscamos que diga CONTINUAR y que offsetParent NO sea null (que no esté oculto)
            if (textoBoton.includes('CONTINUAR') && btn.offsetParent !== null) {
                btnContinuar = btn;
                break;
            }
        }
        
        if (!btnContinuar) throw new Error("No se encontró ningún botón visible que diga CONTINUAR.");

        // Esperar a que se quite el 'disabled'
        let intentos = 0;
        while (btnContinuar.disabled && intentos < 20) {
            await pausa(250);
            intentos++;
        }
        
        if (btnContinuar.disabled) throw new Error("El botón CONTINUAR nunca se habilitó.");

        await pausa(500); 
        
        // --- SÚPER CLIC ---
        btnContinuar.focus(); // Ponemos el foco en el botón
        btnContinuar.click(); // Clic nativo
        btnContinuar.dispatchEvent(new KeyboardEvent('keydown', { bubbles: true, key: 'Enter' })); // Simulamos apretar Enter
        
        console.log("✅ Súper Clic en Continuar ejecutado. Esperando nueva pantalla...");
